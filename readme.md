# Cài đặt yarn và nodejs

# Tải source về máy

git clone https://github.com/truongminhson/Computer-Graphic-CS105.git

# Chuyển vào Folder Computer-Vision

cd Computer-Graphic-CS105

# Cài đặt các package

yarn install

# Chạy chương trình trên localhost:8080

yarn dev

# Build production

yarn build
